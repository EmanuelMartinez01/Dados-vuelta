<script>

// PLAYER

const player = document.getElementById("radioPlayer"); const playBtn = document.getElementById("playBtn"); const navPlay = document.getElementById("navPlay"); const icon = document.getElementById("playIcon"); const status = document.getElementById("liveStatus"); const dot = document.getElementById("liveDot");

const volume = document.getElementById("volumeControl");


function togglePlay(){ if(player.paused){

player.play();

} else {

player.pause();

}

}

playBtn.onclick = togglePlay; navPlay.onclick = togglePlay;


player.addEventListener("play", ()=>{ icon.textContent = "pause"; status.textContent = "EN VIVO"; status.classList.remove("text-gray-400"); status.classList.add("text-[#C1121F]"); dot.classList.remove("bg-gray-500"); dot.classList.add("bg-[#C1121F]");

});


player.addEventListener("pause", ()=>{ icon.textContent = "play_arrow"; status.textContent = "OFFLINE"; status.classList.add("text-gray-400"); status.classList.remove("text-[#C1121F]"); dot.classList.add("bg-gray-500"); dot.classList.remove("bg-[#C1121F]");

});


volume.addEventListener("input", e=>{ player.volume = e.target.value;

});


// FORM

const form = document.getElementById("contactForm"); const msg = document.getElementById("formMsg");


form.addEventListener("submit", (e)=>{ e.preventDefault();


const name = document.getElementById("name").value.trim(); const email = document.getElementById("email").value.trim();

const message = document.getElementById("message").value.trim();


if(!name || !email || !message){ alert("Completá todos los campos"); return;

}


if(!email.includes("@")){ alert("Email inválido"); return;

}


msg.classList.remove("hidden");

form.reset();

});
